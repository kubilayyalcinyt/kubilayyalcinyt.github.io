print("true") # artık eşit bu yüzden true
elif numara1 != numara2:
print("false") # artık eşit bu yüzden false

# bu kısım numara2 += 5 bunun olmadığını varsayarak yapıldı
if numara1 > numara2:
print("true") # numara1 numara 2 den büyük bu yüzden true
elif numara1 < numara2:
