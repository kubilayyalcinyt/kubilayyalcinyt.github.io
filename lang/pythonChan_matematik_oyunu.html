# matematik işlemlerine cevap verme uygulaması ----------------------------------------------------------------------------------
import random
userpuan = 0

while True:

number1 = random.randint(1, 100)
number2 = random.randint(1, 100)
math = random.randint(1, 4)
print("Puan: " + str(userpuan))

print("-"*30 , "\n")
if math == 1:
print(str(number1) + " + " + str(number2) + " = ??")
print("eğer soruyu boş bırakırsan -1 puan alıcaksın")
mathX = input("sorunun cevap:> ")
mathEnd = number1 + number2
if not mathX:
print("bir şey unuttun sanırım...")
userpuan -= 1
elif mathEnd != int(mathX):
print("yanlış bildin sıradaki soru geliyor")
userpuan -= 2
else:
print("doğru bildin")
userpuan += 1
elif math == 2:

print("doğru bildin")
userpuan += 1

print(str(number1) + " / " + str(number2) + " = ??")
