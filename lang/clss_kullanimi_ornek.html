            # class örneği ve self kullanımı
            # ben bunu sanki kendi kütüphanemi yapıyormuşum gibi kodladım
            
            class blackMesa():
                
                def __init__(self, adam, isyeri, yetenekler, kazanc, tkazanc): #buraya self ile hazırladığımız değişkenleri
                    self.adam = adam           # burada self. ile tanımlıyoruz aşşağıda örnekler gibi çekip kullanıyoruz
                    self.isYeri = isyeri
                    self.yetenekler = yetenekler
                    self.kazanc = kazanc
                    self.tkazanc = tkazanc
            
                def calısma(self):
                    self.kazanc += 100
                    print(f"{self.adam} çalıştı ve şu kadar kazandı {self.kazanc}")
            
                def terfi(self):
                    self.tkazanc += 200
                    print(f"{self.adam} terfi Aldı şuandaki maşşı: " + str(self.tkazanc))
                
                def bilgiler(self):
                    print(
                        "-"*10 + "\n”"
                        "Adı:" + self.adam + "\n"
                        + "is Yeri" + self.isYeri + "\n"
                        + "Yetenekler: " + str(self.yetenekler) + "\n"
                        + "Toplam Maaş: " + str(self.tkazanc) + "\n"
                        + "Kazancı: " + str(self.kazanc)
                        + "\n" + "-"*10
                    )
            
            # ikinci class örneği ve ekstraları öncekini al ve yenilerini ekle
            
            class blackMesaBufe(blackMesa): #blackmesa classından bilgileri alıyoruz 
                def __init__(self, adam, isyeri, yetenekler, kazanc, tkazanc, yoneticiGucu):
                    super().__init__(adam, isyeri, yetenekler, kazanc, tkazanc)
                    # super(). ile aynı değerleri almaz ise önceki class için değerler
                    # silinoyor ve burada self eklememize gerek yok
            
                    self.yonetG = yoneticiGucu
                    self.adam = adam
                    self.isYeri = isyeri
                    self.yetenekler = yetenekler
                    self.kazanc = kazanc
                    self.tkazanc = tkazanc
            
                def calısma(self):
                    print(f"{self.adam} çalıştı ve şu kadar kazandı {self.kazanc}")
                    self.kazanc += 100
                    if self.yonetG == 40:
                        self.yonetG += 0.5
            
                def bilgiler(self):
                    super().bilgiler() # aynı mantık super(). ile bilgileri aldık
                    print("Yonetici Gücü: " + str(self.yonetG))
                    if self.yonetG == 100: # ekstra sorgu ekledik
                        print("not: yönetici için doğmuş biri")
                    elif self.yonetG == 70:
                        print("not: belki bir gün yönetici olabilir")
                    else:
                        print("not: bu adam yönetici ise ben amerika nın başkanıyım")

            # tutorial.py
            from tqdm import tqdm
            import time
            
            
            class matbaa():
            
                def __init__(self, murekkeb, islem, kitaplar, para, calismaBitti):
                    self.murekkeb = murekkeb
                    self.islem = islem
                    self.kitaplar = kitaplar = []
                    self.para = para
                    self.calismaBitti = False
                    
                
                def bilgiler(self):
                    print(f"""
                    |----------------------------------------|
                    |Mürekkeb Miktarı: {self.murekkeb}       |
                    |kitaplar: {len(self.kitaplar)}          |
                    |----------------------------------------|
                    |                                        |
                    |                                        |
                    |         --------komutlar--------       |
                    |   1:) mürekkebi doldur                 |
                    |   2:) mürekkeb yardım                  |
                    |   3:) hesap                            |
                    |   4:) kitaplar                         |
                    |   5:) kalan iş                         |
                    ------------------------------------------
                    """)
            
            
                def calis(self):
                    if self.islem == 5:
                        print("çalışılıcak bir şey kalmadı")
                    else:
                        if self.murekkeb <= 0:
                            print("mürekkep kalmadı")
                        else:
                            for isZamani in tqdm(range(2), total=2, position=0, leave=False, ascii=True, desc="çalışılıyor şuanda: "):
                                time.sleep(1)
                            self.islem += 1
                            self.murekkeb -= 1
                            print(str(self.islem) + "kalan iş")
                            self.calismaBitti = True
                            
            
                def kitabıBastır(self):
                    if self.calismaBitti == True:
                            print("işlem bitti kitabın tamamlanması için isim gir \n")
            
                            kitap = input("Kitap Adını Gir: ")
                            self.kitaplar.append(kitap)
                            print("Kitabınızın adı: ", self.kitaplar)
                                
                            self.islem = 0
                            self.para += 150
            
                            print("Kitap tamlandı Şu kadar Kazandın: " + str(self.para))
                            self.calismaBitti = False
                
            
                def murekkebDoldur(self):
                    if self.para == 0:
                        print("paran yok")
                    elif self.para < 100:
                        print("yeterli değil")
                    else:
                        self.murekkeb = 5
                        self.para -= 100
