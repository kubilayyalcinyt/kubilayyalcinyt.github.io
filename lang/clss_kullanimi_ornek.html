    # class örneği ve self kullanımı
    # ben bunu sanki kendi kütüphanemi yapıyormuşum gibi kodladım
    
    class blackMesa():
        
        def __init__(self, adam, isyeri, yetenekler, kazanc, tkazanc): #buraya self ile hazırladığımız değişkenleri
            self.adam = adam           # burada self. ile tanımlıyoruz aşşağıda örnekler gibi çekip kullanıyoruz
            self.isYeri = isyeri
            self.yetenekler = yetenekler
            self.kazanc = kazanc
            self.tkazanc = tkazanc
    
        def calısma(self):
            self.kazanc += 100
            print(f"{self.adam} çalıştı ve şu kadar kazandı {self.kazanc}")
    
        def terfi(self):
            self.tkazanc += 200
            print(f"{self.adam} terfi Aldı şuandaki maşşı: " + str(self.tkazanc))
        
        def bilgiler(self):
            print(
                "-"*10 + "\n”"
                "Adı:" + self.adam + "\n"
                + "is Yeri" + self.isYeri + "\n"
                + "Yetenekler: " + str(self.yetenekler) + "\n"
                + "Toplam Maaş: " + str(self.tkazanc) + "\n"
                + "Kazancı: " + str(self.kazanc)
                + "\n" + "-"*10
            )
    
    # ikinci class örneği ve ekstraları öncekini al ve yenilerini ekle
    
    class blackMesaBufe(blackMesa): #blackmesa classından bilgileri alıyoruz 
        def __init__(self, adam, isyeri, yetenekler, kazanc, tkazanc, yoneticiGucu):
            super().__init__(adam, isyeri, yetenekler, kazanc, tkazanc)
            # super(). ile aynı değerleri almaz ise önceki class için değerler
            # silinoyor ve burada self eklememize gerek yok
    
            self.yonetG = yoneticiGucu
            self.adam = adam
            self.isYeri = isyeri
            self.yetenekler = yetenekler
            self.kazanc = kazanc
            self.tkazanc = tkazanc
    
        def calısma(self):
            print(f"{self.adam} çalıştı ve şu kadar kazandı {self.kazanc}")
            self.kazanc += 100
            if self.yonetG == 40:
                self.yonetG += 0.5
    
        def bilgiler(self):
            super().bilgiler() # aynı mantık super(). ile bilgileri aldık
            print("Yonetici Gücü: " + str(self.yonetG))
            if self.yonetG == 100: # ekstra sorgu ekledik
                print("not: yönetici için doğmuş biri")
            elif self.yonetG == 70:
                print("not: belki bir gün yönetici olabilir")
            else:
                print("not: bu adam yönetici ise ben amerika nın başkanıyım")
