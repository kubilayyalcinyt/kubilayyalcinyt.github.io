# boolean --> true yada false | doğru veya yanlış

# | doğru mu yanlış mı - eşit mi değil mi - büyük mü küçük mü? - içinde var mı yok mu
# true false - == yada != gibi - < yada > | <= yada >= - in yada not in |

# ufak bir örnek --> tabi örnekler sayfasında bolca örnek yazıcam burası bilgi kısmı
uwu = "gnuchan"
# 1234567
# len bize karakter uzunluğunu veriyor
if len(uwu) != 9: # eşit değildir : evet bu yüzden direk bu çalışıcak
print("eşit değil")
elif len(uwu) == 9: # eşittir eğer if konuşu olmaz ise elif olur else if deniyor
print("eşit")
elif len(uwu) > 9:
print("eşit")
else: # en son elif'ler olmaz ise bu devreye girer
print("hiç biri değil")
