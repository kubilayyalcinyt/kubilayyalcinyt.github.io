# dictionary (sözlük) -----------------------------------------------------------------------------------------------------------

oyunPuanları = {
    "half life 3":100,
    "left 4 dead 3":95,
    "portal 3":98,
    "gta 6":98.5,


    "alt oyunlar":{
        "postal 32":62,
        "metin 3":32,
        "var kıraft 3":"yüz puan"

    }
}

# sözlükde anahtar ve değer bulunur anahtar çağrılır değer alınır | ufak not anahtar değerleri int olduğu için str çevirdim
print("Yılın En iyi Oyunu: " + str(oyunPuanları["half life 3"])) # bu şekilde çağırırız yada
#                                                  anahtar
print("Yılın En çöp Oyunu: " + str(oyunPuanları["alt oyunlar"]["metin 3"])) # alt anahtarı çağırdım böyle
#                                                  anahtar     alt anahtar

oyunPuanları["dead island"] = 1000 # yeni değer eklemek
print(oyunPuanları["dead island"])

oyunPuanları["alt oyunlar"]["metin 3"] = 100 # böyle anahtarın değerini değiştirebilirsiniz
print(oyunPuanları["alt oyunlar"]["metin 3"])

oyunPuanları.pop("half life 3") # anahtar silmek için
print(oyunPuanları)

oyunPuanları.popitem() # rasgele anahtar silmek için

yedek = oyunPuanları.copy() # var olan sözlüğü yeni bir sözlük oluşturmak yada olanın içine kopyalamak için
print(yedek)

oyunPuanları.clear() # tamamen silmek için
