import random # rasgele kalime ve sayı yapmak için python kütüphanesi

try: # python da bazen kullanıcı istediğimi şey dışında veri girmeye çalışabiliyor bu yüzden try ve except var
    lowNumber = int(input("En düşük sayı gir: ")) # düşük sayı için input
    highNumber = int(input("En Yüksek Sayı gir: ")) # yüksek sayı için input
    if lowNumber == highNumber: # eğer düşük sayı ve yüksek sayı eşit ise hata veriyor malesef
        highNumber += 1 # bu yüzden +1 yüksek sayı için
except ValueError: # eğer sayı yerine harf girerse hata mesajı veriyoruz
    print("sayı gir --> 1234 gibi falan")
else: # eğer sorun yok ise işlem başlıyor
    number = random.randrange(lowNumber, highNumber) # yüksek sayıyı ve düşük sayı arasında bir sayıyı number için atıyoruz
    #              .randint(lowNumber, highNumber) burada yüksek ve duşuk sayı dahil
    while True:
        guess = int(input("Tahmin Gir: ")) # ve kullanıcı tahmit ediyor tekrar tekrar
        if guess > number:  #yüksek ise tahmin
            print("fazla attın ufak at")
        elif guess < number: # düşük ise tahmin
            print("çok düşük söyledin")
        else: # eğer doğru tahmin ise
            print("evet doğru bildin kazanan numara: " + str(number))
            break # loop bitiyor
